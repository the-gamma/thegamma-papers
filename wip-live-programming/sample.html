Join to country codes

# Set keep_default_na because the Namibia has ISO code NA
df_isos = pd.read_excel('iso.xlsx', keep_default_na=False)
    .drop_duplicates('country_code')

df = df_mat.copy().merge(df_isos, 'left', 'country_code')
   .rename({ 'iso2': 'country_code' }, axis=1)
